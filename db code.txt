create database electronic_store_db ;
CREATE TABLE `usertb1` (
	`Uid` INT NOT NULL AUTO_INCREMENT,

)AUTO_INCREMENT=1;

	`itid` INT NOT NULL AUTO_INCREMENT,

	`itprice` INT(30)NOT NULL ,
	`itcat` VARCHAR(30) NOT NULL ,
